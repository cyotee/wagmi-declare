= Example Contract Lists

This appendix contains complete, working contract list examples for common DeFi patterns.

== Token Swap Interface

A complete token swap form with slippage control and deadline.

[source,json]
----
[
  {
    "chainId": 1,
    "hookName": "UniswapV2Router",
    "name": "Uniswap V2 Swap",
    "functions": [
      {
        "swapExactTokensForTokens": "Swap Tokens",
        "simulate": true,
        "previewConfig": {
          "showTokenTransfers": true,
          "showApprovals": true,
          "warningConditions": [
            {
              "field": "slippage",
              "condition": "gt",
              "value": 500,
              "message": "High slippage may result in unfavorable execution"
            }
          ]
        },
        "gasEstimationConfig": {
          "enabled": true,
          "showInNative": true,
          "showInUsd": true
        },
        "arguments": [
          {
            "name": "tokenIn",
            "type": "address",
            "description": "Token to sell",
            "ui": {
              "widget": "select",
              "source": "tokenlist",
              "sourcePath": "tokens.tokenlist.json",
              "valueField": "address",
              "labelField": "symbol",
              "layout": { "colSpan": 6 }
            }
          },
          {
            "name": "tokenOut",
            "type": "address",
            "description": "Token to buy",
            "ui": {
              "widget": "select",
              "source": "tokenlist",
              "sourcePath": "tokens.tokenlist.json",
              "valueField": "address",
              "labelField": "symbol",
              "layout": { "colSpan": 6 }
            }
          },
          {
            "name": "amountIn",
            "type": "uint256",
            "description": "Amount to sell",
            "ui": {
              "widget": "tokenAmount",
              "tokenAmountConfig": {
                "tokenFrom": "tokenIn",
                "showMaxButton": true,
                "showBalance": true,
                "showUsdValue": true
              },
              "layout": { "colSpan": 12, "emphasis": "prominent" },
              "validation": {
                "onChain": {
                  "abiCall": {
                    "function": "balanceOf",
                    "inlineAbi": [{
                      "name": "balanceOf",
                      "type": "function",
                      "inputs": [{ "name": "account", "type": "address" }],
                      "outputs": [{ "type": "uint256" }]
                    }],
                    "contractFrom": "tokenIn",
                    "argsFrom": ["connectedWallet"]
                  },
                  "condition": "gte",
                  "compareToField": "amountIn",
                  "errorMessage": "Insufficient balance",

            }
          },
          {
            "name": "amountOutMin",
            "type": "uint256",

            "ui": {
              "widget": "tokenAmount",
              "tokenAmountConfig": {
                "tokenFrom": "tokenOut",
                "showUsdValue": true
              },
              "layout": { "readonly": true }
            }
          },
          {
            "name": "slippage",
            "type": "uint256",
            "description": "Slippage tolerance",
            "default": 50,
            "ui": {
              "widget": "slider",
              "validation": {
                "min": 0,
                "max": 1000,
                "step": 10,
                "errorMessage": "Slippage must be 0-1000 bps"
              },
              "display": {
                "unit": "bps",
                "unitLabel": "bps",
                "decimals": 0
              },
              "layout": { "colSpan": 6, "emphasis": "subtle" }
            }
          },
          {
            "name": "deadline",
            "type": "uint256",
            "description": "Transaction deadline",
            "ui": {
              "widget": "datetime",
              "datetimeConfig": {
                "format": "relative",

                "defaultOffset": "+20m"
              },
              "layout": { "colSpan": 6, "emphasis": "subtle" }
            }
          }
        ]
      }
    ]
  }
]

== Liquidity Provision with Wizard

[source,json]
----

              "id": "tokens",

              "id": "amounts",

== Token Vesting Setup

Create vesting schedules with datetime inputs.

[source,json]
----
[
  {
    "chainId": 1,
    "hookName": "VestingFactory",
    "name": "Token Vesting",
    "functions": [
      {
        "createVestingSchedule": "Create Vesting Schedule",
        "arguments": [
          {
            "name": "beneficiary",
            "type": "address",
            "description": "Vesting recipient",
            "ui": {
              "widget": "address",
              "addressBook": true,
              "placeholder": "0x..."
            }
          },
          {
            "name": "token",
            "type": "address",
            "description": "Token to vest",
            "ui": {
              "widget": "select",
              "source": "tokenlist",
              "sourcePath": "tokens.json"
            }
          },
          {
            "name": "totalAmount",
            "type": "uint256",
            "description": "Total amount to vest",
            "ui": {
              "widget": "tokenAmount",
              "tokenAmountConfig": {
                "tokenFrom": "token",
                "showBalance": true,
                "showUsdValue": true
              }
            }
          },
          {
            "name": "startTime",
            "type": "uint256",
            "description": "Vesting start time",
            "ui": {
              "widget": "datetime",
              "datetimeConfig": {
                "format": "datetime",
                "minDate": "now"
              },
              "helpText": "When the vesting period begins"
            }
          },
          {
            "name": "cliffDuration",
            "type": "uint256",
            "description": "Cliff period",
            "ui": {
              "widget": "datetime",
              "datetimeConfig": {
                "format": "relative",
                "defaultOffset": "+180d"
              },
              "helpText": "No tokens vest until after the cliff"
            }
          },
          {
            "name": "vestingDuration",
            "type": "uint256",
            "description": "Total vesting duration",
            "ui": {
              "widget": "datetime",
              "datetimeConfig": {
                "format": "relative",
                "defaultOffset": "+365d"
              },
              "helpText": "Total time from start to fully vested"
            }
          },
          {
            "name": "revocable",
            "type": "bool",
            "description": "Can schedule be revoked?",
            "default": false,
            "ui": {
              "widget": "checkbox",
              "helpText": "If enabled, admin can revoke unvested tokens"
            }
          }
        ]
      }
    ]
  }
]

== Vault Deposit with Protocol Selection

Conditional fields based on selected protocol.

[source,json]
----
[
  {
    "chainId": 1,
    "hookName": "VaultAggregator",
    "name": "Vault Aggregator",
    "functions": [
      {
        "deposit": "Deposit to Vault",
        "arguments": [
          {
            "name": "protocol",
            "type": "string",

              "widget": "select",
              "source": "static",
              "options": [
                { "value": "aave", "label": "Aave V3" },
                { "value": "compound", "label": "Compound V3" },
                { "value": "yearn", "label": "Yearn V3" }

== Multi-Token Approval

Handle array of token approvals.

[source,json]
----
[
  {
    "chainId": 1,
    "hookName": "BatchApprover",
    "name": "Batch Token Approval",
    "functions": [
      {
        "batchApprove": "Approve Multiple Tokens",
        "arguments": [
          {
            "name": "spender",
            "type": "address",
            "description": "Contract to approve",
            "ui": {
              "widget": "address",
              "placeholder": "0x..."
            }
          },
          {
            "name": "tokens",
            "type": "address[]",
            "description": "Tokens to approve",
            "minItems": 1,
            "maxItems": 10,
            "ui": {
              "widget": "multiselect",
              "source": "tokenlist",
              "sourcePath": "tokens.json",
              "array": {
                "addLabel": "Add Token",

                "itemLabelField": "symbol"

          {
            "name": "amounts",
